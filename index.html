    Default Template

    The .twig extension on our Templates means we are able to use Twig tags to render our Template.

    To simply display a variable with Twig, surround the variable name with double curly brackets {{}}.

                    Hello {{ name }}!

    Twig will execute simple logic like if/else, for loops, and variable assignments when surrounded by {% %}

    • Example: {% set condition = 'potato' %}
    • Example: {% if condition == 'yes' %}Conditional Content{% endif %}
    • Reference: http://twig.sensiolabs.org/

                            public function browseItemsAction()
                        {
                            $em = $this->getDoctrine()->getManager();

                            $items = $em->getRepository('MyTeamMyGameBundle:Item')->findAll();

                            $query = $em->createQueryBuilder()
                                ->select('c')
                                ->from('MyTeamMyGameBundle:Item', 'c')
                                ->where('c.name LIKE :prefer')
                                ->setParameters(array('prefer' => 'Armor'))
                                ->setMaxResults(1)
                                ->orderBy("c.value","DESC")
                                ->getQuery();

                            $recommend = $query->getSingleResult();

                            return $this->render('MyTeamMyGameBundle:Default:browseItems.html.twig',
                                array(
                                    'items' => $items,
                                    'recommend' => $recommend
                            ));
                        }

                                $output->writeln("complete.");
                            }
                        }

    Creating a Command 

                                protected function execute(InputInterface $input, OutputInterface $output)
                            {
                                $em = $this->getContainer()->get('doctrine')->getManager('default');
                                $char_id = $input->getArgument('char_id');

                                $character = $em->getRepository('MyTeamMyGameBundle:Characters')
                                                ->findOneById($char_id);

                                $current_health = $character->getCurrentHealth();
                                $max_health = $character->getMaxHealth();

                                $character->setCurrentHealth($max_health);
                                $em->persist($character);
                                $em->flush();

                                $output->writeln("Regeneration Complete.");
                            }

                        public function awardItem(Character $character, Item $item)
                    {
                        $em = $this->em;

                        $char_name = $character->getName();
                        $char_id = $character->getId();
                        $item_name = $item->getName();
                        $item_id = $item->getId();

                        $owned = $em->getRepository('MyTeamMyGameBundle:Storage')
                                    ->findOneBy(array(
                                        'item_id'=>$item_id,
                                        'char_id'=>$char_id
                                    ));
                        if (empty($owned)){
                            $owned = new Storage();
                            $owned->setItemId($item_id);
                            $owned->setCharId($char_id);
                            $owned->setAmount(0);
                        }
                        $owned->setAmount($owned->getAmount() + 1);
                        $em->persist($owned);
                        $em->flush();

                        $message = "$char_name has received a new $item_name!";

                        return $message;
                    }
